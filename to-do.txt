@GetMapping("/me")
public ResponseEntity<?> me(@AuthenticationPrincipal Jwt jwt) {
    return ResponseEntity.ok(Map.of(
        "userId", jwt.getSubject(),
        "email", jwt.getClaim("email"),
        "roles", jwt.getClaim("roles")
    ));
}
